🧠 排序算法总览表#

🔍 各算法核心思想讲解与代码示例#

1️⃣ 冒泡排序 Bubble Sort#

每一轮两两比较，把最大值“冒泡”到最后。

func bubbleSort(nums []int) {
	n := len(nums)
	for i := 0; i < n-1; i++ {
		for j := 0; j < n-i-1; j++ {
			if nums[j] > nums[j+1] {
				nums[j], nums[j+1] = nums[j+1], nums[j]
			}
		}
	}
}

✅ 稳定、实现简单 ❌ 慢，不适合大数据

2️⃣ 选择排序 Selection Sort#

每轮选出最小值放到前面。

func selectionSort(nums []int) {
	n := len(nums)
	for i := 0; i < n-1; i++ {
		min := i
		for j := i + 1; j < n; j++ {
			if nums[j] < nums[min] {
				min = j
			}
		}
		nums[i], nums[min] = nums[min], nums[i]
	}
}

❌ 不稳定（因为最小值和前面元素交换会破坏顺序） ✅ 简单、交换次数少

3️⃣ 插入排序 Insertion Sort#

把当前元素插入到前面已排序的部分。

func insertionSort(nums []int) {
	for i := 1; i < len(nums); i++ {
		key := nums[i]
		j := i - 1
		for j >= 0 && nums[j] > key {
			nums[j+1] = nums[j]
			j--
		}
		nums[j+1] = key
	}
}

✅ 稳定，小规模或几乎有序数组很快。 📈 平均 O(n²)，最好 O(n)。

4️⃣ 希尔排序 Shell Sort#

插入排序 + 分组加速。

func shellSort(nums []int) {
	for gap := len(nums) / 2; gap > 0; gap /= 2 {
		for i := gap; i < len(nums); i++ {
			temp := nums[i]
			j := i
			for ; j >= gap && nums[j-gap] > temp; j -= gap {
				nums[j] = nums[j-gap]
			}
			nums[j] = temp
		}
	}
}

❌ 不稳定 ✅ 比插入排序快得多。

5️⃣ 归并排序 Merge Sort#

分治 + 合并（见你前面的模板）。 ✅ 稳定 ❌ 需要 O(n) 额外空间。

6️⃣ 快速排序 Quick Sort#

选一个基准，把小于它的放左边，大于的放右边，再递归。

func quickSort(nums []int) {
	if len(nums) <= 1 {
		return
	}
	sort(nums, 0, len(nums)-1)
}

func sort(nums []int, l, r int) {
	if l >= r {
		return
	}
	pivot := partition(nums, l, r)
	sort(nums, l, pivot-1)
	sort(nums, pivot+1, r)
}

func partition(nums []int, l, r int) int {
	pivot := nums[r]
	i := l
	for j := l; j < r; j++ {
		if nums[j] < pivot {
			nums[i], nums[j] = nums[j], nums[i]
			i++
		}
	}
	nums[i], nums[r] = nums[r], nums[i]
	return i
}

⚡ 通常是最快的内部排序算法。 ❌ 不稳定，最坏情况 O(n²)，但平均非常好。

7️⃣ 堆排序 Heap Sort#

构建最大堆，每次取出堆顶（最大值）放到末尾。

func heapSort(nums []int) {
	n := len(nums)
	for i := n/2 - 1; i >= 0; i-- {
		heapify(nums, n, i)
	}
	for i := n - 1; i > 0; i-- {
		nums[0], nums[i] = nums[i], nums[0]
		heapify(nums, i, 0)
	}
}

func heapify(nums []int, n, i int) {
	largest := i
	l, r := 2*i+1, 2*i+2
	if l < n && nums[l] > nums[largest] {
		largest = l
	}
	if r < n && nums[r] > nums[largest] {
		largest = r
	}
	if largest != i {
		nums[i], nums[largest] = nums[largest], nums[i]
		heapify(nums, n, largest)
	}
}

✅ 不需要额外空间，O(1) 空间复杂度 ❌ 不稳定

8️⃣ 计数排序 Counting Sort#

统计每个元素出现次数，按次数输出。

func countingSort(nums []int, maxVal int) []int {
	count := make([]int, maxVal+1)
	for _, v := range nums {
		count[v]++
	}
	idx := 0
	for i, c := range count {
		for c > 0 {
			nums[idx] = i
			idx++
			c--
		}
	}
	return nums
}

✅ 稳定、线性时间 ❌ 只能用于非负整数，且范围不能太大。

9️⃣ 桶排序 Bucket Sort#

把数据分进多个桶，每个桶内部再排序。

func bucketSort(nums []float64) []float64 {
	n := len(nums)
	buckets := make([][]float64, n)
	for _, v := range nums {
		idx := int(v * float64(n))
		buckets[idx] = append(buckets[idx], v)
	}
	
	// 对每个桶内部排序（可用插入排序）
	for i := range buckets {
		insertionSortFloat(buckets[i])
	}
	
	result := make([]float64, 0, n)
	for _, b := range buckets {
		result = append(result, b...)
	}
	return result
}

func insertionSortFloat(nums []float64) {
	for i := 1; i < len(nums); i++ {
		key := nums[i]
		j := i - 1
		for j >= 0 && nums[j] > key {
			nums[j+1] = nums[j]
			j--
		}
		nums[j+1] = key
	}
}

适用于：分布均匀的浮点数（例如 [0,1) 区间内）。

🔟 基数排序 Radix Sort#

按个位、十位、百位……依次使用计数排序。

func radixSort(nums []int) {
	maxVal := 0
	for _, v := range nums {
		if v > maxVal {
			maxVal = v
		}
	}
	for exp := 1; maxVal/exp > 0; exp *= 10 {
		countingSortByDigit(nums, exp)
	}
}

func countingSortByDigit(nums []int, exp int) {
	n := len(nums)
	output := make([]int, n)
	count := make([]int, 10)

	for i := 0; i < n; i++ {
		digit := (nums[i] / exp) % 10
		count[digit]++
	}
	for i := 1; i < 10; i++ {
		count[i] += count[i-1]
	}
	for i := n - 1; i >= 0; i-- {
		digit := (nums[i] / exp) % 10
		output[count[digit]-1] = nums[i]
		count[digit]--
	}
	copy(nums, output)
}

适用于：整数、定长字符串。

📈 总结#

类别 算法 口诀
O(n²) 类 冒泡、选择、插入 慢但简单，冒泡插入稳，选择不稳
O(n log n) 类 归并、快排、堆排 分治三兄弟：归并稳，快排快，堆排省空间
线性排序类 计数、桶、基数 利用数据特征换取速度，空间换时间
